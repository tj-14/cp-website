  cout << "cmath examples\n";
  cout << "abs(-5): " << abs(-5) << "\n";
  cout << "pow(2, 3): " << pow(2, 3) << "\n";
  cout << "sqrt(64): " << sqrt(64) << "\n";
  cout << "log2(64): " << log2(64) << "\n";
  cout << "exp(3): " << exp(3) << "\n";
  cout << "\n";

  cout << "algorithm examples\n";
  int a = 4, b = 7;
  cout << "a b: " << a << " " << b << "\n";
  swap(a, b);
  cout << "a b: " << a << " " << b << "\n";
  cout << "min(a, b): " << min(a, b) << "\n";
  cout << "max(a, b): " << max(a, b) << "\n";
  cout << "__gcd(a, b): " << __gcd(a, b) << "\n";
}
