#include <bits/stdc++.h>

using namespace std;

void printVec(vector<int> v) {
  for (int i = 0; i < v.size(); i++) {
    cout << v[i] << " ";
  }
  cout << "\n";
}

int main() {
  vector<int> A;
  A.push_back(2);
  A.push_back(5);
  printVec(A); // OUTPUT: 2 5 \n

  A.push_back(9);
  printVec(A); // OUTPUT: 2 5 9 \n

  cout << A.front() << "\n"; // OUTPUT: 2\n
  cout << A.back() << "\n";  // OUTPUT: 9\n
  cout << A.size() << "\n";  // OUTPUT: 3\n

  A.pop_back();
  A.pop_back();
  printVec(A); // OUTPUT: 2 \n
}
